11function parse(query) {
12 let ppos = RE_PARAM.exec(query);
13 let curpos = 0;
14 let start = 0;
15 let end;
16 const parts = [];
17 let inQuote = false;
18 let escape = false;
19 let qchr;
20 const tokens = [];
21 let qcnt = 0;
22 let lastTokenEndPos = 0;
23 let i;
24
25 if (ppos) {
26 do {
27 for (i=curpos,end=ppos.index; i<end; ++i) {
28 let chr = query.charCodeAt(i);
29 if (chr === BSLASH)
30 escape = !escape;
31 else {
32 if (escape) {
33 escape = false;
34 continue;
35 }
36 if (inQuote && chr === qchr) {
37 if (query.charCodeAt(i + 1) === qchr) {
38 // quote escaped via "" or ''
39 ++i;
40 continue;
41 }
42 inQuote = false;
43 } else if (chr === DQUOTE || chr === SQUOTE) {
44 inQuote = true;
45 qchr = chr;
46 }
47 }
48 }
49 if (!inQuote) {
50 parts.push(query.substring(start, end));
51 tokens.push(ppos[0].length === 1 ? qcnt++ : ppos[1]);
52 start = end + ppos[0].length;
53 lastTokenEndPos = start;
54 }
55 curpos = end + ppos[0].length;
56 } while (ppos = RE_PARAM.exec(query));
57
58 if (tokens.length) {
59 if (curpos < query.length) {
60 parts.push(query.substring(lastTokenEndPos));
61 }
62 return [parts, tokens];
63 }
64 }
65 return [query];
66};

109 function join(tree) {
110 if (tree.length == 1) {
111 return tree;
112 }
113
114 let unnamed = noTailingSemicolon(tree[0][0]);
115 for (let i=1; i < tree[0].length; ++i) {
116 if (tree[0][i-1].slice(-1) == ':') {
117 unnamed += config.placeholder;
118 }
119 unnamed += config.placeholder;
120 unnamed += noTailingSemicolon(tree[0][i]);
121 }
122
123 const last = tree[0][tree[0].length -1];
124 if (tree[0].length == tree[1].length) {
125 if (last.slice(-1) == ':') {
126 unnamed += config.placeholder;
127 }
128 unnamed += config.placeholder;
129 }
130 return [unnamed, tree[1]];
