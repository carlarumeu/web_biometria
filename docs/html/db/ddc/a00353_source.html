18function leftPad(num, value) {
19 const s = value.toString();
20 // if we don't need to pad
21 if (s.length >= num) {
22 return s;
23 }
24 return (pad + s).slice(-num);
25}

27// The whole reason parse* function below exist
28// is because String creation is relatively expensive (at least with V8), and if we have
29// a buffer with "12345" content ideally we would like to bypass intermediate
30// "12345" string creation and directly build 12345 number out of
31// <Buffer 31 32 33 34 35> data.
32// In my benchmarks the difference is ~25M 8-digit numbers per second vs
33// 4.5 M using Number(packet.readLengthCodedString())
34// not used when size is close to max precision as series of *10 accumulate error
35// and approximate result mihgt be diffreent from (approximate as well) Number(bigNumStringValue))
36// In the futire node version if speed difference is smaller parse* functions might be removed
37// don't consider them as Packet public API

200 readLengthCodedNumberExt(tag, bigNumberStrings, signed) {
201 let word0, word1;
202 let res;
203 if (tag === 0xfb) {
204 return null;
205 }
206 if (tag === 0xfc) {
207 return this.readInt8() + (this.readInt8() << 8);
208 }
209 if (tag === 0xfd) {
210 return this.readInt8() + (this.readInt8() << 8) + (this.readInt8() << 16);
211 }
212 if (tag === 0xfe) {
213 // TODO: check version
214 // Up to MySQL 3.22, 0xfe was followed by a 4-byte integer.
215 word0 = this.readInt32();
216 word1 = this.readInt32();
217 if (word1 === 0) {
218 return word0; // don't convert to float if possible
219 }
220 if (word1 < 2097152) {
221 // max exact float point int, 2^52 / 2^32
222 return word1 * 0x100000000 + word0;
223 }
224 res = new Long(word0, word1, !signed); // Long need unsigned
225 const resNumber = res.toNumber();
226 const resString = res.toString();
227 res = resNumber.toString() === resString ? resNumber : resString;
228 return bigNumberStrings ? resString : res;
229 }
230 // eslint-disable-next-line no-console
231 console.trace();
232 throw new Error(`Should not reach here: ${tag}`);

255 // DATE, DATETIME and TIMESTAMP
256 readDateTime(timezone) {
257 if (!timezone || timezone === 'Z' || timezone === 'local') {
258 const length = this.readInt8();
259 if (length === 0xfb) {
260 return null;
261 }
262 let y = 0;
263 let m = 0;
264 let d = 0;
265 let H = 0;
266 let M = 0;
267 let S = 0;
268 let ms = 0;
269 if (length > 3) {
270 y = this.readInt16();
271 m = this.readInt8();
272 d = this.readInt8();
273 }
274 if (length > 6) {
275 H = this.readInt8();
276 M = this.readInt8();
277 S = this.readInt8();
278 }
279 if (length > 10) {
280 ms = this.readInt32() / 1000;
281 }
282 // NO_ZERO_DATE mode and NO_ZERO_IN_DATE mode are part of the strict
283 // default SQL mode used by MySQL 8.0. This means that non-standard
284 // dates like '0000-00-00' become NULL. For older versions and other
285 // possible MySQL flavours we still need to account for the
286 // non-standard behaviour.
287 if (y + m + d + H + M + S + ms === 0) {
288 return INVALID_DATE;
289 }
290 if (timezone === 'Z') {
291 return new Date(Date.UTC(y, m - 1, d, H, M, S, ms));
292 }
293 return new Date(y, m - 1, d, H, M, S, ms);
294 }
295 let str = this.readDateTimeString(6, 'T');
296 if (str.length === 10) {
297 str += 'T00:00:00';
298 }
299 return new Date(str + timezone);
300 }

302 readDateTimeString(decimals, timeSep) {
303 const length = this.readInt8();
304 let y = 0;
305 let m = 0;
306 let d = 0;
307 let H = 0;
308 let M = 0;
309 let S = 0;
310 let ms = 0;
311 let str;
312 if (length > 3) {
313 y = this.readInt16();
314 m = this.readInt8();
315 d = this.readInt8();
316 str = [leftPad(4, y), leftPad(2, m), leftPad(2, d)].join('-');
317 }
318 if (length > 6) {
319 H = this.readInt8();
320 M = this.readInt8();
321 S = this.readInt8();
322 str += `${timeSep || ' '}${[
323 leftPad(2, H),
324 leftPad(2, M),
325 leftPad(2, S)
326 ].join(':')}`;
327 }
328 if (length > 10) {
329 ms = this.readInt32();
330 str += '.';
331 if (decimals) {
332 ms = leftPad(6, ms);
333 if (ms.length > decimals) {
334 ms = ms.substring(0, decimals); // rounding is done at the MySQL side, only 0 are here
335 }
336 }
337 str += ms;
338 }
339 return str;
340 }

342 // TIME - value as a string, Can be negative
343 readTimeString(convertTtoMs) {
344 const length = this.readInt8();
345 if (length === 0) {
346 return '00:00:00';
347 }
348 const sign = this.readInt8() ? -1 : 1; // 'isNegative' flag byte
349 let d = 0;
350 let H = 0;
351 let M = 0;
352 let S = 0;
353 let ms = 0;
354 if (length > 6) {
355 d = this.readInt32();
356 H = this.readInt8();
357 M = this.readInt8();
358 S = this.readInt8();
359 }
360 if (length > 10) {
361 ms = this.readInt32();
362 }
363 if (convertTtoMs) {
364 H += d * 24;
365 M += H * 60;
366 S += M * 60;
367 ms += S * 1000;
368 ms *= sign;
369 return ms;
370 }
371 // Format follows mySQL TIME format ([-][h]hh:mm:ss[.u[u[u[u[u[u]]]]]])
372 // For positive times below 24 hours, this makes it equal to ISO 8601 times
373 return (
374 (sign === -1 ? '-' : '') +
375 [leftPad(2, d * 24 + H), leftPad(2, M), leftPad(2, S)].join(':') +
376 (ms ? `.${ms}`.replace(/0+$/, '') : '')
377 );
378 }

433 parseInt(len, supportBigNumbers) {
434 if (len === null) {
435 return null;
436 }
437 if (len >= 14 && !supportBigNumbers) {
438 const s = this.buffer.toString('ascii', this.offset, this.offset + len);
439 this.offset += len;
440 return Number(s);
441 }
442 let result = 0;
443 const start = this.offset;
444 const end = this.offset + len;
445 let sign = 1;
446 if (len === 0) {
447 return 0; // TODO: assert? exception?
448 }
449 if (this.buffer[this.offset] === minus) {
450 this.offset++;
451 sign = -1;
452 }
453 // max precise int is 9007199254740992
454 let str;
455 const numDigits = end - this.offset;
456 if (supportBigNumbers) {
457 if (numDigits >= 15) {
458 str = this.readString(end - this.offset, 'binary');
459 result = parseInt(str, 10);
460 if (result.toString() === str) {
461 return sign * result;
462 }
463 return sign === -1 ? `-${str}` : str;
464 }
465 if (numDigits > 16) {
466 str = this.readString(end - this.offset);
467 return sign === -1 ? `-${str}` : str;
468 }
469 }
470 if (this.buffer[this.offset] === plus) {
471 this.offset++; // just ignore
472 }
473 while (this.offset < end) {
474 result *= 10;
475 result += this.buffer[this.offset] - 48;
476 this.offset++;
477 }
478 const num = result * sign;
479 if (!supportBigNumbers) {
480 return num;
481 }
482 str = this.buffer.toString('ascii', start, end);
483 if (num.toString() === str) {
484 return num;
485 }
486 return str;
487 }

525 function parseGeometry() {
526 let x, y, i, j, numPoints, line;
527 let result = null;
528 const byteOrder = buffer.readUInt8(offset);
529 offset += 1;
530 const wkbType = byteOrder
531 ? buffer.readUInt32LE(offset)
532 : buffer.readUInt32BE(offset);
533 offset += 4;
534 switch (wkbType) {
535 case 1: // WKBPoint
536 x = byteOrder
537 ? buffer.readDoubleLE(offset)
538 : buffer.readDoubleBE(offset);
539 offset += 8;
540 y = byteOrder
541 ? buffer.readDoubleLE(offset)
542 : buffer.readDoubleBE(offset);
543 offset += 8;
544 result = { x: x, y: y };
545 break;
546 case 2: // WKBLineString
547 numPoints = byteOrder
548 ? buffer.readUInt32LE(offset)
549 : buffer.readUInt32BE(offset);
550 offset += 4;
551 result = [];
552 for (i = numPoints; i > 0; i--) {
553 x = byteOrder
554 ? buffer.readDoubleLE(offset)
555 : buffer.readDoubleBE(offset);
556 offset += 8;
557 y = byteOrder
558 ? buffer.readDoubleLE(offset)
559 : buffer.readDoubleBE(offset);
560 offset += 8;
561 result.push({ x: x, y: y });
562 }
563 break;
564 case 3: // WKBPolygon
565 // eslint-disable-next-line no-case-declarations
566 const numRings = byteOrder
567 ? buffer.readUInt32LE(offset)
568 : buffer.readUInt32BE(offset);
569 offset += 4;
570 result = [];
571 for (i = numRings; i > 0; i--) {
572 numPoints = byteOrder
573 ? buffer.readUInt32LE(offset)
574 : buffer.readUInt32BE(offset);
575 offset += 4;
576 line = [];
577 for (j = numPoints; j > 0; j--) {
578 x = byteOrder
579 ? buffer.readDoubleLE(offset)
580 : buffer.readDoubleBE(offset);
581 offset += 8;
582 y = byteOrder
583 ? buffer.readDoubleLE(offset)
584 : buffer.readDoubleBE(offset);
585 offset += 8;
586 line.push({ x: x, y: y });
587 }
588 result.push(line);
589 }
590 break;
591 case 4: // WKBMultiPoint
592 case 5: // WKBMultiLineString
593 case 6: // WKBMultiPolygon
594 case 7: // WKBGeometryCollection
595 // eslint-disable-next-line no-case-declarations
596 const num = byteOrder
597 ? buffer.readUInt32LE(offset)
598 : buffer.readUInt32BE(offset);
599 offset += 4;
600 result = [];
601 for (i = num; i > 0; i--) {
602 result.push(parseGeometry());
603 }
604 break;
605 }
606 return result;
607 }
608 return parseGeometry();
609 }

658 parseFloat(len) {
659 if (len === null) {
660 return null;
661 }
662 let result = 0;
663 const end = this.offset + len;
664 let factor = 1;
665 let pastDot = false;
666 let charCode = 0;
667 if (len === 0) {
668 return 0; // TODO: assert? exception?
669 }
670 if (this.buffer[this.offset] === minus) {
671 this.offset++;
672 factor = -1;
673 }
674 if (this.buffer[this.offset] === plus) {
675 this.offset++; // just ignore
676 }
677 while (this.offset < end) {
678 charCode = this.buffer[this.offset];
679 if (charCode === dot) {
680 pastDot = true;
681 this.offset++;
682 } else if (charCode === exponent || charCode === exponentCapital) {
683 this.offset++;
684 const exponentValue = this.parseInt(end - this.offset);
685 return (result / factor) * Math.pow(10, exponentValue);
686 } else {
687 result *= 10;
688 result += this.buffer[this.offset] - 48;
689 this.offset++;
690 if (pastDot) {
691 factor = factor * 10;
692 }
693 }
694 }
695 return result / factor;
696 }
