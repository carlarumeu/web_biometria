187 static mergeFlags(default_flags, user_flags) {
188 let flags = 0x0,
189 i;
190 if (!Array.isArray(user_flags)) {
191 user_flags = String(user_flags || '')
192 .toUpperCase()
193 .split(/\s*,+\s*/);
194 }
195 // add default flags unless "blacklisted"
196 for (i in default_flags) {
197 if (user_flags.indexOf(`-${default_flags[i]}`) >= 0) {
198 continue;
199 }
200 flags |= ClientConstants[default_flags[i]] || 0x0;
201 }
202 // add user flags unless already already added
203 for (i in user_flags) {
204 if (user_flags[i][0] === '-') {
205 continue;
206 }
207 if (default_flags.indexOf(user_flags[i]) >= 0) {
208 continue;
209 }
210 flags |= ClientConstants[user_flags[i]] || 0x0;
211 }
212 return flags;
213 }

262 static parseUrl(url) {
263 const parsedUrl = new URL(url);
264 const options = {
265 host: decodeURIComponent(parsedUrl.hostname),
266 port: parseInt(parsedUrl.port, 10),
267 database: decodeURIComponent(parsedUrl.pathname.slice(1)),
268 user: decodeURIComponent(parsedUrl.username),
269 password: decodeURIComponent(parsedUrl.password),
270 };
271 parsedUrl.searchParams.forEach((value, key) => {
272 try {
273 // Try to parse this as a JSON expression first
274 options[key] = JSON.parse(value);
275 } catch (err) {
276 // Otherwise assume it is a plain string
277 options[key] = value;
278 }
279 });
280 return options;
281 }
