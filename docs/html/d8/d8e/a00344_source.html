226 let length = 17; // = 4 padding + 1 + 12 for the rest
227 fields.forEach(field => {
228 length += Packet.lengthCodedStringLength(
229 column[field],
230 CharsetToEncoding[column.characterSet]

232 });

235 const packet = new Packet(sequenceId, buffer, 0, length);
236 function writeField(name) {
237 packet.writeLengthCodedString(
238 column[name],
239 CharsetToEncoding[column.characterSet]
240 );
241 }
242 packet.offset = 4;
243 fields.forEach(writeField);
244 packet.writeInt8(0x0c);
245 packet.writeInt16(column.characterSet);
246 packet.writeInt32(column.columnLength);
247 packet.writeInt8(column.columnType);
248 packet.writeInt16(column.flags);
249 packet.writeInt8(column.decimals);
250 packet.writeInt16(0); // filler
251 return packet;
252 }
