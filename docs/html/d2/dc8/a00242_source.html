35function authSwitchRequest(packet, connection, command) {
36 const { pluginName, pluginData } = Packets.AuthSwitchRequest.fromPacket(
37 packet
38 );
39 let authPlugin =
40 connection.config.authPlugins && connection.config.authPlugins[pluginName];
41
42 // legacy plugin api don't allow to override mysql_native_password
43 // if pluginName is mysql_native_password it's using standard auth4.1 auth
44 if (
45 connection.config.authSwitchHandler &&
46 pluginName !== 'mysql_native_password'
47 ) {
48 const legacySwitchHandler = connection.config.authSwitchHandler;
49 warnLegacyAuthSwitch();
50 legacySwitchHandler({ pluginName, pluginData }, (err, data) => {
51 if (err) {
52 return authSwitchPluginError(err, command);
53 }
54 connection.writePacket(new Packets.AuthSwitchResponse(data).toPacket());
55 });
56 return;
57 }
58 if (!authPlugin) {
59 authPlugin = standardAuthPlugins[pluginName];
60 }
61 if (!authPlugin) {
62 throw new Error(
63 `Server requests authentication using unknown plugin ${pluginName}. See ${'TODO: add plugins doco here'} on how to configure or author authentication plugins.`
64 );
65 }
66 connection._authPlugin = authPlugin({ connection, command });
67 Promise.resolve(connection._authPlugin(pluginData)).then(data => {
68 if (data) {
69 connection.writePacket(new Packets.AuthSwitchResponse(data).toPacket());
70 }
71 }).catch(err => {
72 authSwitchPluginError(err, command);
73 });
74}

76function authSwitchRequestMoreData(packet, connection, command) {

78
79 if (connection.config.authSwitchHandler) {

81 warnLegacyAuthSwitch();
