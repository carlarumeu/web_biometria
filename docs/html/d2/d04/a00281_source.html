212 // notify all commands in the queue and bubble error as connection "error"
213 // called on stream error or unexpected termination
214 _notifyError(err) {
215 if (this.connectTimeout) {
216 Timers.clearTimeout(this.connectTimeout);
217 this.connectTimeout = null;
218 }
219 // prevent from emitting 'PROTOCOL_CONNECTION_LOST' after EPIPE or ECONNRESET
220 if (this._fatalError) {
221 return;
222 }
223 let command;
224 // if there is no active command, notify connection
225 // if there are commands and all of them have callbacks, pass error via callback
226 let bubbleErrorToConnection = !this._command;
227 if (this._command && this._command.onResult) {
228 this._command.onResult(err);
229 this._command = null;
230 // connection handshake is special because we allow it to be implicit
231 // if error happened during handshake, but there are others commands in queue
232 // then bubble error to other commands and not to connection
233 } else if (
234 !(
235 this._command &&
236 this._command.constructor === Commands.ClientHandshake &&
237 this._commands.length > 0
238 )
239 ) {
240 bubbleErrorToConnection = true;
241 }
242 while ((command = this._commands.shift())) {
243 if (command.onResult) {
244 command.onResult(err);
245 } else {
246 bubbleErrorToConnection = true;
247 }
248 }
249 // notify connection if some comands in the queue did not have callbacks
250 // or if this is pool connection ( so it can be removed from pool )
251 if (bubbleErrorToConnection || this._pool) {
252 this.emit('error', err);
253 }
254 // close connection after emitting the event in case of a fatal error
255 if (err.fatal) {
256 this.close();
257 }
258 }

293 writePacket(packet) {
294 const MAX_PACKET_LENGTH = 16777215;
295 const length = packet.length();
296 let chunk, offset, header;
297 if (length < MAX_PACKET_LENGTH) {
298 packet.writeHeader(this.sequenceId);
299 if (this.config.debug) {
300 // eslint-disable-next-line no-console
301 console.log(
302 `${this._internalId} ${this.connectionId} <== ${this._command._commandName}#${this._command.stateName()}(${[this.sequenceId, packet._name, packet.length()].join(',')})`
303 );
304 // eslint-disable-next-line no-console
305 console.log(
306 `${this._internalId} ${this.connectionId} <== ${packet.buffer.toString('hex')}`
307 );
308 }
309 this._bumpSequenceId(1);
310 this.write(packet.buffer);
311 } else {
312 if (this.config.debug) {
313 // eslint-disable-next-line no-console
314 console.log(
315 `${this._internalId} ${this.connectionId} <== Writing large packet, raw content not written:`
316 );
317 // eslint-disable-next-line no-console
318 console.log(
319 `${this._internalId} ${this.connectionId} <== ${this._command._commandName}#${this._command.stateName()}(${[this.sequenceId, packet._name, packet.length()].join(',')})`
320 );
321 }
322 for (offset = 4; offset < 4 + length; offset += MAX_PACKET_LENGTH) {
323 chunk = packet.buffer.slice(offset, offset + MAX_PACKET_LENGTH);
324 if (chunk.length === MAX_PACKET_LENGTH) {
325 header = Buffer.from([0xff, 0xff, 0xff, this.sequenceId]);
326 } else {
327 header = Buffer.from([
328 chunk.length & 0xff,
329 (chunk.length >> 8) & 0xff,
330 (chunk.length >> 16) & 0xff,
331 this.sequenceId
332 ]);
333 }
334 this._bumpSequenceId(1);
335 this.write(header);
336 this.write(chunk);
337 }
338 }

400 protocolError(message, code) {
401 // Starting with MySQL 8.0.24, if the client closes the connection
402 // unexpectedly, the server will send a last ERR Packet, which we can
403 // safely ignore.
404 // https://dev.mysql.com/worklog/task/?id=12999
405 if (this._closing) {
406 return;
407 }
408
409 const err = new Error(message);
410 err.fatal = true;
411 err.code = code || 'PROTOCOL_ERROR';
412 this.emit('error', err);

419 handlePacket(packet) {
420 if (this._paused) {
421 this._paused_packets.push(packet);
422 return;
423 }
424 if (this.config.debug) {
425 if (packet) {
426 // eslint-disable-next-line no-console
427 console.log(
428 ` raw: ${packet.buffer
429 .slice(packet.offset, packet.offset + packet.length())
430 .toString('hex')}`
431 );
432 // eslint-disable-next-line no-console
433 console.trace();
434 const commandName = this._command
435 ? this._command._commandName
436 : '(no command)';
437 const stateName = this._command
438 ? this._command.stateName()
439 : '(no command)';
440 // eslint-disable-next-line no-console
441 console.log(
442 `${this._internalId} ${this.connectionId} ==> ${commandName}#${stateName}(${[packet.sequenceId, packet.type(), packet.length()].join(',')})`
443 );
444 }
445 }
446 if (!this._command) {
447 const marker = packet.peekByte();
448 // If it's an Err Packet, we should use it.
449 if (marker === 0xff) {
450 const error = Packets.Error.fromPacket(packet);
451 this.protocolError(error.message, error.code);
452 } else {
453 // Otherwise, it means it's some other unexpected packet.
454 this.protocolError(
455 'Unexpected packet while no commands in the queue',
456 'PROTOCOL_UNEXPECTED_PACKET'
457 );
458 }
459 this.close();
460 return;
461 }
462 if (packet) {
463 // Note: when server closes connection due to inactivity, Err packet ER_CLIENT_INTERACTION_TIMEOUT from MySQL 8.0.24, sequenceId will be 0
464 if (this.sequenceId !== packet.sequenceId) {
465 const err = new Error(
466 `Warning: got packets out of order. Expected ${this.sequenceId} but received ${packet.sequenceId}`
467 );
468 err.expected = this.sequenceId;
469 err.received = packet.sequenceId;
470 this.emit('warn', err); // REVIEW
471 // eslint-disable-next-line no-console
472 console.error(err.message);
473 }
474 this._bumpSequenceId(packet.numPackets);
475 }
476 try {
477 if (this._fatalError) {
478 // skip remaining packets after client is in the error state
479 return;
480 }
481 const done = this._command.execute(packet, this);
482 if (done) {
483 this._command = this._commands.shift();
484 if (this._command) {
485 this.sequenceId = 0;
486 this.compressedSequenceId = 0;
487 this.handlePacket();
488 }
489 }
490 } catch (err) {
491 this._handleFatalError(err);
492 this.stream.destroy();
493 }
494 }

620 execute(sql, values, cb) {
621 let options = {
622 infileStreamFactory: this.config.infileStreamFactory
623 };
624 if (typeof sql === 'object') {
625 // execute(options, cb)
626 options = {
627 ...options,
628 ...sql,
629 sql: sql.sql,
630 values: sql.values
631 };
632 if (typeof values === 'function') {
633 cb = values;
634 } else {
635 options.values = options.values || values;
636 }
637 } else if (typeof values === 'function') {
638 // execute(sql, cb)
639 cb = values;
640 options.sql = sql;
641 options.values = undefined;
642 } else {
643 // execute(sql, values, cb)
644 options.sql = sql;
645 options.values = values;
646 }
647 this._resolveNamedPlaceholders(options);
648 // check for values containing undefined
649 if (options.values) {
650 //If namedPlaceholder is not enabled and object is passed as bind parameters
651 if (!Array.isArray(options.values)) {
652 throw new TypeError(
653 'Bind parameters must be array if namedPlaceholders parameter is not enabled'
654 );
655 }
656 options.values.forEach(val => {
657 //If namedPlaceholder is not enabled and object is passed as bind parameters
658 if (!Array.isArray(options.values)) {
659 throw new TypeError(
660 'Bind parameters must be array if namedPlaceholders parameter is not enabled'
661 );
662 }
663 if (val === undefined) {
664 throw new TypeError(
665 'Bind parameters must not contain undefined. To pass SQL NULL specify JS null'
666 );
667 }
668 if (typeof val === 'function') {
669 throw new TypeError(
670 'Bind parameters must not contain function(s). To pass the body of a function as a string call .toString() first'
671 );
672 }
673 });
674 }
675 const executeCommand = new Commands.Execute(options, cb);
676 const prepareCommand = new Commands.Prepare(options, (err, stmt) => {
677 if (err) {
678 // skip execute command if prepare failed, we have main
679 // combined callback here
680 executeCommand.start = function() {
681 return null;
682 };
683 if (cb) {
684 cb(err);
685 } else {
686 executeCommand.emit('error', err);
687 }
688 executeCommand.emit('end');
689 return;
690 }
691 executeCommand.statement = stmt;
692 });
693 this.addCommand(prepareCommand);
694 this.addCommand(executeCommand);
695 return executeCommand;
696 }
