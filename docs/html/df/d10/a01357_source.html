196 // invalid type
197 if (expected === false) {
198 return false
199 }
200
201 // split types
202 var actualParts = actual.split('/')
203 var expectedParts = expected.split('/')
204
205 // invalid format
206 if (actualParts.length !== 2 || expectedParts.length !== 2) {
