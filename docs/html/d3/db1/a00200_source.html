399 if (string.indexOf('::') !== string.lastIndexOf('::')) {
400 return null;
401 }

403 if (zoneId) {
404 zoneId = zoneId.substring(1);
405 string = string.replace(/%.+$/, '');
406 }
407 colonCount = 0;
408 lastColon = -1;
409 while ((lastColon = string.indexOf(':', lastColon + 1)) >= 0) {
410 colonCount++;
411 }
412 if (string.substr(0, 2) === '::') {
413 colonCount--;
414 }
415 if (string.substr(-2, 2) === '::') {
416 colonCount--;
417 }
