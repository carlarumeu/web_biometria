78 readParameter(packet, connection) {
79 // there might be scenarios when mysql server reports more parameters than
80 // are actually present in the array of parameter definitions.
81 // if EOF packet is received we switch to "read fields" state if there are
82 // any fields reported by the server, otherwise we finish the command.
83 if (packet.isEOF()) {
84 if (this.fieldCount > 0) {
85 return Prepare.prototype.readField;
86 }
87 return this.prepareDone(connection);
88 }
89 const def = new Packets.ColumnDefinition(packet, connection.clientEncoding);
90 this.parameterDefinitions.push(def);
91 if (this.parameterDefinitions.length === this.parameterCount) {
92 return Prepare.prototype.parametersEOF;
93 }
94 return this.readParameter;
95 }
96
97 readField(packet, connection) {
98 if (packet.isEOF()) {
99 return this.prepareDone(connection);
100 }
101 const def = new Packets.ColumnDefinition(packet, connection.clientEncoding);
102 this.fields.push(def);
103 if (this.fields.length === this.fieldCount) {
104 return Prepare.prototype.fieldsEOF;
105 }
106 return Prepare.prototype.readField;
107 }

109 parametersEOF(packet, connection) {
110 if (!packet.isEOF()) {
