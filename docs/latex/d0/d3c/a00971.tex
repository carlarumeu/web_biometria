\hypertarget{a00971_autotoc_md2230}{}\doxysubsection{\texorpdfstring{Features}{Features}}\label{a00971_autotoc_md2230}

\begin{DoxyItemize}
\item Works both in \href{https://nodejs.org}{\texttt{ Node.\+js}} and in the browser
\item If a number is supplied to {\ttfamily ms}, a string with a unit is returned
\item If a string that contains the number is supplied, it returns it as a number (e.\+g.\+: it returns {\ttfamily 100} for `\textquotesingle{}100'\`{})
\item If you pass a string with a number and a valid unit, the number of equivalent milliseconds is returned
\end{DoxyItemize}\hypertarget{a00971_autotoc_md2231}{}\doxysubsection{\texorpdfstring{Related Packages}{Related Packages}}\label{a00971_autotoc_md2231}
